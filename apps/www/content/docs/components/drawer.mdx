## Setup

If you don't already have the snippet, run the following command to add the
`drawer` snippet

```sh
npx @chakra-ui/cli snippet add drawer
```

The snippet includes a closed component composition for the `Drawer` component.

## Usage

```jsx
import {
  DrawerBackdrop,
  DrawerBody,
  DrawerCloseTrigger,
  DrawerContent,
  DrawerFooter,
  DrawerHeader,
  DrawerRoot,
  DrawerTitle,
  DrawerTrigger,
} from "@/components/ui/drawer"
```

```jsx
<DrawerRoot>
  <DrawerBackdrop />
  <DrawerTrigger />
  <DrawerContent>
    <DrawerCloseTrigger />
    <DrawerHeader>
      <DrawerTitle />
    </DrawerHeader>
    <DrawerBody />
    <DrawerFooter />
  </DrawerContent>
</DrawerRoot>
```
