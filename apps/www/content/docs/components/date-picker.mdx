## Usage

```tsx
import { DatePicker } from "@chakra-ui/react"
```

```tsx
<DatePicker.Root>
  {data.map((item) => (
    <DatePicker.Item key={item.label}>
      <DatePicker.ItemLabel>{item.label}</DatePicker.ItemLabel>
      <DatePicker.ItemValue>{item.value}</DatePicker.ItemValue>
    </DatePicker.Item>
  ))}
</DatePicker.Root>
```
