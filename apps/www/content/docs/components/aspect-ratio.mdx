## Usage

```jsx
import { AspectRatio } from "@chakra-ui/react"
```

```jsx
<AspectRatio>
  <iframe
    title="naruto"
    src="https://www.youtube.com/embed/QhBnZ6NPOY0"
    allowFullScreen
  />
</AspectRatio>
```
