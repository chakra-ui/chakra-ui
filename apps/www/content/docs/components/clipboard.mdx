## Usage

```jsx
import { Clipboard } from "@chakra-ui/react"
```

```jsx
<Clipboard.Root>
  <Clipboard.Control>
    <Clipboard.Input />
    <Clipboard.Trigger />
  </Clipboard.Control>
</Clipboard.Root>
```
